Using for loops and dictionaries.

dict = {'Alpha': 1, 'Beta': 2}

for key in dict.keys():
    print (key)                     # Alpha Beta

for value in dict.values():
    print (value)                   # 1 2

for key, value in dict.items():
    print (key, value)              # ('Alpha', 1) ('Beta', 2)
